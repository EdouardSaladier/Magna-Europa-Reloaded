BUL_historical_requirements_met = {
	OR = {
		has_war_with = SOV
		any_allied_country = {
			has_war_with = SOV
		}
	}
	SOV = {
		surrender_progress < 0.05
	}
	OR = {
		SOV = {
			any_controlled_state = {
				OR = {
					has_state_flag = BUL_initial_bulgarian_state_flag
					has_state_flag = BUL_historical_bulgarian_claim_flag
				}
				is_fully_controlled_by = PREV
			}
		}
		SOV = {
			any_controlled_state = {
				is_core_of = GER
				NOT = {
					OR = {
						is_core_of = ESP
						is_core_of = LIT
						is_core_of = POL
					}
				}
			}
		}
	}
}
